#!/bin/bash

FIP=`cat /var/log/secure |awk '/Failed/{print $(NF-3)}'| sort |uniq -c |awk '{print $2"="$1;}'`
for i in $FIP
do
  IP=`echo $i  |awk -F '=' '{print $1}'`
  NUM=`echo $i |awk -F '=' '{print $2}'`
  if [ ${#NUM} -gt 1 ]; then
    grep $IP /etc/hosts.deny > /dev/null
    if [ $? -gt 0 ]; then
      echo "sshd:$IP:deny" >> /etc/hosts.deny
      echo "$(date '+%Y-%m-%d %H:%M:%S') $IP" >> /var/log/block-ip.log
    fi
  fi
done
